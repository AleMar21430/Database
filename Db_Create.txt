CREATE TABLE IF NOT EXISTS Medico(
	id SERIAL PRIMARY KEY,
	nombre TEXT NOT NULL,
	apellido TEXT NOT NULL,
	telefono TEXT,
	direccion TEXT,
	numero_colegiado INTEGER NOT NULL,
	especialidad TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS Paciente(
	id SERIAL PRIMARY KEY,
	nombre TEXT NOT NULL,
	apellido TEXT NOT NULL,
	telefono TEXT,
	direccion TEXT,
	peso REAL,
	altura REAL,
	imc REAL
);

CREATE TABLE IF NOT EXISTS Precedentes_Paciente(
	paciente_id INTEGER,
	precedentes_enfermedad TEXT,
	FOREIGN KEY (paciente_id) REFERENCES Paciente (id) ON DELETE CASCADE
	ON UPDATE CASCADE
);

CREATE TABLE IF NOT EXISTS Adiccion(
	paciente_id INTEGER,
	nombre TEXT,
	descripcion TEXT,
	FOREIGN KEY (paciente_id) REFERENCES Paciente (id) ON DELETE CASCADE
	ON UPDATE CASCADE
);

CREATE TABLE IF NOT EXISTS Examen(
	paciente_id INTEGER,
	nombre TEXT,
	FOREIGN KEY (paciente_id) REFERENCES Paciente (id) ON DELETE CASCADE
	ON UPDATE CASCADE
);

CREATE TABLE IF NOT EXISTS Enfermedad(
	paciente_id INTEGER,
	nombre TEXT,
	FOREIGN KEY (paciente_id) REFERENCES Paciente (id) ON DELETE CASCADE
	ON UPDATE CASCADE
);

CREATE TABLE IF NOT EXISTS Cirugia(
	paciente_id INTEGER,
	nombre TEXT,
	FOREIGN KEY (paciente_id) REFERENCES Paciente (id) ON DELETE CASCADE
	ON UPDATE CASCADE
);

CREATE TABLE IF NOT EXISTS Afeccion(
	id SERIAL PRIMARY KEY,
	organo_afectado TEXT
);

CREATE TABLE IF NOT EXISTS Medicamento(
	id SERIAL PRIMARY KEY,
	nombre TEXT,
	descripcion TEXT,
	afeccion_id INTEGER,
	FOREIGN KEY (afeccion_id) REFERENCES Afeccion (id) ON DELETE CASCADE
	ON UPDATE CASCADE
);

CREATE TABLE IF NOT EXISTS Centro_Salud(
	id SERIAL PRIMARY KEY,
	nombre TEXT,
	localizacion TEXT,
	medicamento_id INTEGER,
	FOREIGN KEY (medicamento_id) REFERENCES Paciente (id) ON DELETE CASCADE
	ON UPDATE CASCADE
);

CREATE TABLE IF NOT EXISTS Inventario(
	centro_salud_id INTEGER,
	medicamento_id INTEGER,
	porcentaje_por_vencer INTEGER NULL,
	cantidad_existencias INTEGER,
	FOREIGN KEY (centro_salud_id) REFERENCES Centro_salud (id) ON DELETE CASCADE
	ON UPDATE CASCADE,
	FOREIGN KEY (medicamento_id) REFERENCES Medicamento (id) ON DELETE CASCADE
	ON UPDATE CASCADE
);

CREATE TABLE IF NOT EXISTS Trabajo(
	fecha_entarada DATE NOT NULL,
	fecha_salida DATE NULL,
	medico_id INTEGER,
	centro_salud_id INTEGER,
	FOREIGN KEY (medico_id) REFERENCES Medico (id) ON DELETE CASCADE
	ON UPDATE CASCADE,
	FOREIGN KEY (centro_salud_id) REFERENCES Centro_salud (id) ON DELETE CASCADE
	ON UPDATE CASCADE
);

CREATE TABLE IF NOT EXISTS Historial_Medico(
	fecha DATE NOT NULL,
	diagnostico TEXT,
	tratamiento TEXT,
	observaciones TEXT,
	status TEXT,
	medico_id INTEGER,
	paciente_id INTEGER,
	centro_salud_id INTEGER,
	FOREIGN KEY (centro_salud_id) REFERENCES Centro_salud (id) ON DELETE CASCADE
	ON UPDATE CASCADE,
	FOREIGN KEY (paciente_id) REFERENCES Paciente (id) ON DELETE CASCADE
	ON UPDATE CASCADE,
	FOREIGN KEY (medico_id) REFERENCES Medico (id) ON DELETE CASCADE
	ON UPDATE CASCADE
);

CREATE TABLE IF NOT EXISTS Credenciales (
	id SERIAL PRIMARY KEY,
	usuario TEXT,
	contrasenia TEXT,
	tipo TEXT
);
