SELECT hm.diagnostico, COUNT(hm.diagnostico) AS cantidad_casos, hm.status
FROM historial_medico  hm
WHERE LOWER(hm.status) = 'fallecido'
GROUP BY hm.diagnostico, hm.status
ORDER BY COUNT(hm.diagnostico)
LIMIT 10;
